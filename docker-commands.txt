  --name my-postgres-container \
  --network backend-network \

  -e DB_HOST=basic3tier \
  -p 5432:5432 \
  postgres:latest

  --name my-node-api-container \
  --network backend-network \
  -e DB_HOST=my-postgres-container \
  -p 7000:3000 \
  my-node-api
